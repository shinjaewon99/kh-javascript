            연산 코드;
            [return 결과값;]
        }

        function func1(x){

        }

익명 함수


        var f2 = function([매개변수[, 매개변수 [, ...]]]){
            연산코드;
            [return [결과값];]
        }

        //실행
        f2([전달인자[, 전달인자[, ...]]]);

예제 코드


        function func2(x){
            var f2 =function(){
                return 10;
            };
            var result = x + f2();
            return result;
        }

호출없이 바로 실행하는 함수

        }

예제 코드


        function func3(x){
            var y =function(){
                return 10;
            }();

        }

자바스크립트 함수의 매개변수 처리

자바 스크립트의 함수는 전달인자의 수와 매개변수의 수가 동일하지 않아도 함수가 동작되도록 구성되어있다.

매개 변수의 수보다 전달인자의 수가 많은 경우에는 argument 배열에 순차적으로 저장되며 ,매개 변수의 수보다 전달인자의 수가 적은 경우에는 매개변수에 전달되지 않은 인자에 대해 undefined 로 처리가 된다.

팁 : argument 배열에는 모든 전달인자가 저장되어 있다.

예제 코드


        function fun4(x,y){

        }
